

## 使用方式一

```tsx
@Extend(Column) function CustomerPackageManageStyle() {
  .width('100%')
  .backgroundColor(Color.White)
  .borderRadius(10)
  .alignItems(HorizontalAlign.Start)
  .margin({top:10})
  .padding({top:13.5,left:12,bottom:13.5,right:12,})
}

@Component
export  default struct CustomerPackageManage{ // 套餐管理

  @Prop sectionIds:string = '';
  @Prop packageList:PackageItem[] = [];
  @Prop dataList:DatalistModel
  @Prop carGoodsDate: CustomerResponseModel[];


  build() {
    Column() {
      Text('套餐管理').fontColor('#323232').fontSize(14).fontWeight(FontWeight.Bold)

      packageCard1({
        dataList:this.dataList
      })
      packageCard2({
        carGoodsDate: this.carGoodsDate
      })

    }
    .CustomerPackageManageStyle()
    .id(this.sectionIds) // 正确设置单个ID
  }



}
```
