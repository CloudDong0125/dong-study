@Link @Watch('onIsActiveTime') isActiveTime: number;
@State startTime: string = Util.DateUtils.getOneMonthAgoDate();
@State endTime: string = Util.DateUtils.getCurrentDate();

 private onIsActiveTime() {
    this.onIsActiveTimeChange()
  }
onIsActiveTimeChange() {

    switch (this.isActiveTime) {
      case 0:
        this.startTime = Util.DateUtils.getYesterdayDate()
        this.endTime = Util.DateUtils.getYesterdayDate()
        break;
      case 1:
        this.startTime = Util.DateUtils.getCurrentDate()
        this.endTime = Util.DateUtils.getCurrentDate()
        break;
      case 2:
        this.startTime = Util.DateUtils.getStartOfMonth()
        this.endTime = Util.DateUtils.getCurrentDate()
        break;
      default:
        this.startTime = Util.DateUtils.getCurrentDate()
        this.endTime = Util.DateUtils.getCurrentDate()
        return;

    }
    console.log('comSearch-tabIndex',this.isActiveTime, this.startTime,this.endTime)
  }

build() {
    Column() {
      Text(this.startTime)
      Text(this.endTime)
    }
  	.onClick(() => {
     		this.showPanel = true
    })       
}
@Builder /* 弹框 */
  myBuilder() {
    Text('单据筛选').fontWeight(FontWeight.Bold).fontSize(17)
    Row() {
            Text("开始时间：")
              .fontColor("#5a5f63")
              .fontSize(12)
            Text(startTime)
              .textStyleInput()
              .fontColor(startTime.startsWith("选择") ? "#999" : "#333")
              .onClick(() => {
                this.showDatePicker('start')
              })
          }

          Row() {
            Text("结束时间：")
              .fontColor("#5a5f63")
              .fontSize(12)
            Text(endTime)
              .textStyleInput()
              .fontColor(endTime.startsWith("选择") ? "#999" : "#333")
              .onClick(() => {
                this.showDatePicker('end')
              })
          }
  }
